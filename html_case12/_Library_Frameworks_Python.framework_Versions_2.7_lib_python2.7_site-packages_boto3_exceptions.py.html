  33. # We're subclassing from botocore.exceptions.DataNotFoundError
  34. # to keep backwards compatibility with anyone that was catching
  35. # this low level Botocore error before this exception was
  36. # introduced in boto3.
  37. # Same thing for ResourceNotExistsError below.
  38. class UnknownAPIVersionError(Boto3Error,
  39.                              botocore.exceptions.DataNotFoundError):
  40.     def __init__(self, service_name, bad_api_version,
  41.                  available_api_versions):
  42.         msg = (
  43.             "The '%s' resource does not an API version of: %s\n"
  44.             "Valid API versions are: %s"
  45.             % (service_name, bad_api_version, available_api_versions)
  46.         )
  47.         # Not using super because we don't want the DataNotFoundError
  48.         # to be called, it has a different __init__ signature.
  49.         Boto3Error.__init__(self, msg)

  56.         msg = (
  57.             "The '%s' resource does not exist.\n"
  58.             "The available resources are:\n"
  59.             "   - %s\n" % (service_name, '\n   - '.join(available_services))
  60.         )
  61.         if has_low_level_client:
  62.             msg += (
  63.                 "\nConsider using a boto3.client('%s') instead "

  65.         # Not using super because we don't want the DataNotFoundError
  66.         # to be called, it has a different __init__ signature.
  67.         Boto3Error.__init__(self, msg)
