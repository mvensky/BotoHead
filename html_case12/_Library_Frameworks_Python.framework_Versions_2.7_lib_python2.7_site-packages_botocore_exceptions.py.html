 346. class ClientError(Exception):
 347.     MSG_TEMPLATE = (
 348.         'An error occurred ({error_code}) when calling the {operation_name} '
 349.         'operation{retry_info}: {error_message}')
 350. 
 351.     def __init__(self, error_response, operation_name):
 352.         retry_info = self._get_retry_info(error_response)
 353.         error = error_response.get('Error', {})
 354.         msg = self.MSG_TEMPLATE.format(
 355.             error_code=error.get('Code', 'Unknown'),
 356.             error_message=error.get('Message', 'Unknown'),
 357.             operation_name=operation_name,
 358.             retry_info=retry_info,
 359.         )
 360.         super(ClientError, self).__init__(msg)
 361.         self.response = error_response
 362.         self.operation_name = operation_name
 363. 
 364.     def _get_retry_info(self, response):
 365.         retry_info = ''
 366.         if 'ResponseMetadata' in response:
 367.             metadata = response['ResponseMetadata']
 368.             if metadata.get('MaxAttemptsReached', False):
 369.                 if 'RetryAttempts' in metadata:
 370.                     retry_info = (' (reached max retries: %s)' %
 371.                                   metadata['RetryAttempts'])
 372.         return retry_info
